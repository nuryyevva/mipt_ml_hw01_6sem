15def compute_accuracy(preds: torch.Tensor, targets: torch.Tensor) -> torch.Tensor: 

16 """ 

17 Compute accuracy given predictions and targets. 

18 

19 :param preds: A tensor containing the predicted class labels (indices of the maximum logits). 

20 :param targets: A tensor containing the ground truth class labels. 

21 

22 :return: A scalar tensor containing the accuracy (ratio of correct predictions to total predictions). 

23 """ 

24 result = (targets == preds).float().mean() 

25 return result

110def train_one_batch( 

111 model: nn.Module, 

112 images: torch.Tensor, 

113 labels: torch.Tensor, 

114 criterion: nn.Module, 

115 optimizer: torch.optim.Optimizer, 

116 device: torch.device, 

117) -> torch.Tensor: 

118 """ 

119 Train the model on one batch of data. 

120 

121 :param model: The neural network model to be trained. 

122 :param images: A batch of input images. 

123 :param labels: A batch of ground truth labels corresponding to the images. 

124 :param criterion: The loss function used to compute the loss. 

125 :param optimizer: The optimizer used to update the model's parameters. 

126 :param device: The device (e.g., CPU or GPU) on which the computation is performed. 

127 

128 :return: The computed loss for the batch. 

129 """ 

130 images = images.to(device) 

131 labels = labels.to(device) 

132 

133 outputs = model(images) # Forward pass: compute predictions 

134 loss = criterion(outputs, labels) # Compute the loss 

135 

136 loss.backward() # Backward pass: compute gradients 

137 optimizer.step() # Update model parameters 

138 optimizer.zero_grad() # Clear gradients for the next iteration 

139 

140 return loss

143def evaluate( 

144 model: nn.Module, test_loader: torch.utils.data.DataLoader, device: torch.device 

145) -> torch.Tensor: 

146 """ 

147 Evaluate the model on the test dataset. 

148 

149 :param model: The neural network model to be evaluated. 

150 :param test_loader: The DataLoader for the test dataset. 

151 :param device: The device (e.g., CPU or GPU) on which the computation is performed. 

152 

153 :return: A scalar tensor containing the accuracy of the model on the test dataset. 

154 """ 

155 all_preds = [] # Store all predictions 

156 all_labels = [] # Store all ground truth labels 

157 

158 for test_images, test_labels in test_loader: 

159 test_images = test_images.to(device) 

160 test_labels = test_labels.to(device) 

161 

162 with torch.inference_mode(): # Disable gradient computation for evaluation 

163 outputs = model(test_images) # Forward pass: compute predictions 

164 preds = torch.argmax( 

165 outputs, 1 

166 ) # Get the predicted class labels (indices of the maximum logits) 

167 

168 all_preds.append(preds) 

169 all_labels.append(test_labels) 

170 

171 # Concatenate all predictions and labels, then compute accuracy 

172 accuracy = compute_accuracy(torch.cat(all_preds), torch.cat(all_labels)) 

173 return accuracy

176def train() -> None: 

177 """ 

178 Main function to run the training pipeline. 

179 

180 Initializes Wandb, loads datasets, trains the model, evaluates it periodically, 

181 and saves the trained model and Wandb run ID. 

182 """ 

183 train_dataset, test_dataset = get_datasets() 

184 train_loader, test_loader = get_dataloaders( 

185 train_dataset, test_dataset, config["batch_size"] 

186 ) 

187 

188 device = torch.device("cpu") 

189 model = get_model(device) # Initialize the model 

190 

191 criterion = ( 

192 nn.CrossEntropyLoss() 

193 ) # Define the loss function (CrossEntropyLoss for classification) 

194 optimizer = torch.optim.AdamW( 

195 model.parameters(), 

196 lr=config["learning_rate"], 

197 weight_decay=config["weight_decay"], 

198 ) # Define the optimizer (AdamW with learning rate and weight decay from config) 

199 

200 for epoch in trange(config["epochs"]): # Loop over the number of epochs 

201 for i, (images, labels) in enumerate( 

202 tqdm(train_loader) 

203 ): # Loop over batches in the training DataLoader 

204 loss = train_one_batch(model, images, labels, criterion, optimizer, device) 

205 

206 # Evaluate the model every 100 batches 

207 if i % 100 == 0: 

208 accuracy = evaluate( 

209 model, test_loader, device 

210 ) # Compute accuracy on the test set 

211 metrics = {"test_acc": accuracy, "train_loss": loss} # Log metrics 

212 wandb.log( 

213 metrics, 

214 step=epoch * len(train_dataset) 

215 + (i + 1) * config["batch_size"], # Log step in Wandb 

216 ) 

217 

218 torch.save(model.state_dict(), "model.pt") # Save the trained model 

219 

220 with open("run_id.txt", "w+") as f: 

221 print(wandb.run.id, file=f) # Save the Wandb run ID to a file 

222

224if __name__ == "__main__": 

225 train()
